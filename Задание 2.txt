        private void BtnOk_Click(object sender, RoutedEventArgs e)
        {
            int num = int.Parse(firstMean.Text);
            if (num >  0)
                num -= 8;
            else if (num < 0)
                num += 6;
            else
               num = 10;

            Answer.Text = $"{num}";
        }
